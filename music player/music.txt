        const uploadBtn           = document.getElementById('upload-btn');
        const uploadAreaPanel     = document.getElementById('upload-area');
        const closeUploadBtn      = document.getElementById('close-upload');
        const fileInput           = document.getElementById('file-input');
        const dropZone            = document.getElementById('drop-zone');
        const clearPlaylistBtn    = document.getElementById('clear-playlist');
        const playAllBtn          = document.getElementById('play-all');
        const shufflePlaylistBtn  = document.getElementById('shuffle-playlist');
        const autoplayToggle      = document.getElementById('autoplay-toggle');
        const lyricsBtn           = document.getElementById('lyrics-btn');
        const lyricsModal         = document.getElementById('lyrics-modal');
        const closeLyricsBtn      = document.getElementById('close-lyrics');
        const lyricsContent       = document.getElementById('lyrics-content');
        const visualizerToggle    = document.getElementById('visualizer-toggle');
        const waveBars            = document.querySelectorAll('.wave-bar');
        const equalizerBars       = document.querySelectorAll('.equalizer-bar');
        const albumArtImage       = document.getElementById('album-art-image');
        const albumArtPlaceholder = document.getElementById('album-art-placeholder');
        const themeToggle         = document.getElementById('theme-toggle');
        const speedSlider         = document.getElementById('speed-slider');
        const playbackSpeedEl     = document.getElementById('playback-speed');
 
        // ─── Helpers ──────────────────────────────────────────────────────────────
        function formatTime(seconds) {
            if (!isFinite(seconds) || isNaN(seconds)) return '0:00';
            const mins = Math.floor(seconds / 60);
            const secs = Math.floor(seconds % 60);
            return `${mins}:${secs < 10 ? '0' : ''}${secs}`;
        }
 
        function formatBytes(bytes) {
            if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)} KB`;
            return `${(bytes / (1024 * 1024)).toFixed(2)} MB`;
        }
 
        function randomColor() {
            const palette = ['#8b5cf6','#10b981','#f59e0b','#3b82f6','#ec4899','#ef4444','#06b6d4','#84cc16'];
            return palette[Math.floor(Math.random() * palette.length)];
        }
 
        // ─── Web Audio API init (lazy, after user gesture) ────────────────────────
        function initAudioContext() {
            if (audioContext) return;
            try {
                audioContext = new (window.AudioContext || window.webkitAudioContext)();
                analyser = audioContext.createAnalyser();
                analyser.fftSize = 256;
                mediaSource = audioContext.createMediaElementSource(audioPlayer);
                mediaSource.connect(analyser);
                analyser.connect(audioContext.destination);
            } catch (e) {
                console.warn('Web Audio API not available:', e);
            }
        }
 
        // ─── UI update: now-playing info ──────────────────────────────────────────
        function updateSongInfo() {
            if (currentSongIndex < 0 || playlist.length === 0) {
                songTitleEl.textContent = 'Select a Song';
                songArtistEl.textContent = 'Artist Name';
                songAlbumEl.textContent = 'Album Name';
                totalDurationEl.textContent = '0:00';
                currentTimeEl.textContent = '0:00';
                albumArtPlaceholder.classList.remove('hidden');
                albumArtImage.classList.add('hidden');
                return;
            }
 
            const song = playlist[currentSongIndex];
            songTitleEl.textContent = song.title;
            songArtistEl.textContent = song.artist;
            songAlbumEl.textContent = song.album;
 
            // Album art: use embedded image if available, else colour gradient
            if (song.coverUrl) {
                albumArtImage.src = song.coverUrl;
                albumArtImage.classList.remove('hidden');
                albumArtPlaceholder.classList.add('hidden');
            } else {
                albumArtImage.classList.add('hidden');
                albumArtPlaceholder.classList.remove('hidden');
                albumArtPlaceholder.style.background =
                    `linear-gradient(135deg, ${song.color} 0%, ${song.color}80 100%)`;
            }
 
            // Duration shown only after metadata loads
            totalDurationEl.textContent = song.durationFormatted || '0:00';
 
            // Lyrics
            lyricsContent.innerHTML = song.lyrics
                ? `<pre class="text-center font-mono text-player-text leading-loose">${song.lyrics}</pre>`
                : `<p class="text-center text-player-text-secondary py-8">Currenty no lyrics available for this song</p>`;
 
            // Progress bar reset
            progressBar.value = 0;
            progressBar.style.setProperty('--progress', '0%');
        }
 
        // ─── Playlist render ──────────────────────────────────────────────────────
        function updatePlaylist() {
            playlistItemsEl.innerHTML = '';
 
            if (playlist.length === 0) {
                emptyPlaylistEl.classList.remove('hidden');
                updateStats();
                return;
            }
 
            emptyPlaylistEl.classList.add('hidden');
 
            playlist.forEach((song, index) => {
                const item = document.createElement('div');
                item.className = `playlist-item p-3 rounded-lg cursor-pointer transition-all duration-300 ${index === currentSongIndex ? 'active' : ''}`;
                item.innerHTML = `
                    <div class="flex items-center justify-between">
                        <div class="flex items-center space-x-3">
                            <div class="w-10 h-10 rounded-lg flex items-center justify-center flex-shrink-0" style="background: linear-gradient(135deg, ${song.color} 0%, ${song.color}80 100%)">
                                <i class="fas fa-music text-white/80"></i>
                            </div>
                            <div class="min-w-0">
                                <h4 class="font-medium truncate">${song.title}</h4>
                                <p class="text-sm text-player-text-secondary truncate">${song.artist}</p>
                            </div>
                        </div>
                        <div class="flex items-center space-x-3 flex-shrink-0 ml-2">
                            <span class="text-sm text-player-text-secondary font-mono">${song.durationFormatted || '--:--'}</span>
                            <button class="p-2 rounded-full hover:bg-player-border transition-colors" data-index="${index}">
                                <i class="fas fa-play text-xs"></i>
                            </button>
                        </div>
                    </div>
                `;
                item.addEventListener('click', () => playSong(index));
                playlistItemsEl.appendChild(item);
            });
 
            updateStats();
        }
 
        // ─── Library stats ────────────────────────────────────────────────────────
        function updateStats() {
            const count = playlist.length;
            const totalSeconds = playlist.reduce((sum, s) => sum + (s.durationSeconds || 0), 0);
            const storageMB = (totalStorageBytes / (1024 * 1024)).toFixed(2);
            const storagePct = Math.min((totalStorageBytes / (500 * 1024 * 1024)) * 100, 100).toFixed(1);
 
            playlistCountEl.textContent = `${count} song${count !== 1 ? 's' : ''}`;
            totalSongsEl.textContent = count;
            totalDurationStatsEl.textContent = formatTime(totalSeconds);
            storageUsedEl.textContent = `${storageMB} MB`;
            storagePercentageEl.textContent = `${storagePct}%`;
            storageBarEl.style.width = `${storagePct}%`;
        }
 
        // ─── Core: load & play ────────────────────────────────────────────────────
        function playSong(index) {
            if (index < 0 || index >= playlist.length) return;
 
            // Resume AudioContext if suspended (browser autoplay policy)
            if (audioContext && audioContext.state === 'suspended') {
                audioContext.resume();
            }
 
            currentSongIndex = index;
            const song = playlist[currentSongIndex];
 
            // Revoke previous object URL to avoid memory leaks
            if (audioPlayer._objectUrl && audioPlayer._objectUrl !== song.src) {
                // keep current song's URL alive, just clear the reference
            }
 
            audioPlayer.src = song.src;
            audioPlayer.load();
 
            updateSongInfo();
            updatePlaylist();
 
            // Always play when a song is explicitly selected
            audioPlayer.play().then(() => {
                isPlaying = true;
                setPlayingUI(true);
                startVisualizerIfReady();
            }).catch(err => {
                console.warn('Playback failed:', err);
            });
        }
 
        // ─── Play/Pause ───────────────────────────────────────────────────────────
        function togglePlayPause() {
            if (playlist.length === 0) return;
 
            // If nothing is loaded yet, load the first song
            if (currentSongIndex < 0) {
                playSong(0);
                return;
            }
 
            initAudioContext();
 
            if (isPlaying) {
                audioPlayer.pause();
                isPlaying = false;
                setPlayingUI(false);
                stopVisualizer();
            } else {
                audioPlayer.play().then(() => {
                    isPlaying = true;
                    setPlayingUI(true);
                    startVisualizerIfReady();
                }).catch(err => console.warn('Playback failed:', err));
            }
        }
 
        function setPlayingUI(playing) {
            playPauseIcon.className = playing ? 'fas fa-pause text-2xl' : 'fas fa-play text-2xl';
            fabPlayIcon.className   = playing ? 'fas fa-pause text-black' : 'fas fa-play text-black';
            // Animate equalizer bars in header
            equalizerBars.forEach(b => b.classList.toggle('paused', !playing));
        }
 
        // ─── Next / Prev ──────────────────────────────────────────────────────────
        function playNext() {
            if (playlist.length === 0) return;
            const next = isShuffled
                ? Math.floor(Math.random() * playlist.length)
                : (currentSongIndex + 1) % playlist.length;
            playSong(next);
        }
 
        function playPrev() {
            if (playlist.length === 0) return;
            // If more than 3 seconds in, restart; otherwise go to previous
            if (audioPlayer.currentTime > 3) {
                audioPlayer.currentTime = 0;
                return;
            }
            const prev = isShuffled
                ? Math.floor(Math.random() * playlist.length)
                : (currentSongIndex - 1 + playlist.length) % playlist.length;
            playSong(prev);
        }
 
        // ─── Shuffle / Repeat ─────────────────────────────────────────────────────
        function toggleShuffle() {
            isShuffled = !isShuffled;
            shuffleBtn.querySelector('i').className = isShuffled
                ? 'fas fa-random text-player-primary'
                : 'fas fa-random text-player-text-secondary';
            shuffleBtn.classList.toggle('bg-player-primary/20', isShuffled);
        }
 
        function toggleRepeat() {
            isRepeating = !isRepeating;
            repeatBtn.querySelector('i').className = isRepeating
                ? 'fas fa-redo text-player-primary'
                : 'fas fa-redo text-player-text-secondary';
            repeatBtn.classList.toggle('bg-player-primary/20', isRepeating);
            audioPlayer.loop = isRepeating;
        }
 
        // ─── Progress bar ─────────────────────────────────────────────────────────
        function updateProgressBar() {
            if (!audioPlayer.duration || !isFinite(audioPlayer.duration)) return;
            const pct = (audioPlayer.currentTime / audioPlayer.duration) * 100;
            progressBar.value = pct;
            progressBar.style.setProperty('--progress', `${pct}%`);
            currentTimeEl.textContent = formatTime(audioPlayer.currentTime);
        }
 
        function seekAudio() {
            if (!audioPlayer.duration || !isFinite(audioPlayer.duration)) return;
            audioPlayer.currentTime = (progressBar.value / 100) * audioPlayer.duration;
        }
 
        // ─── Volume ───────────────────────────────────────────────────────────────
        function updateVolume() {
            audioPlayer.volume = volumeSlider.value / 100;
            volumeSlider.style.setProperty('--volume', `${volumeSlider.value}%`);
        }
 
        // ─── Visualizer ───────────────────────────────────────────────────────────
        function startVisualizerIfReady() {
            if (!analyser) return;
            startVisualizer();
        }
 
        function startVisualizer() {
            if (!analyser) { initAudioContext(); return; }
            if (animationId) cancelAnimationFrame(animationId);
 
            const bufferLength = analyser.frequencyBinCount;
            const dataArray = new Uint8Array(bufferLength);
 
            function animate() {
                animationId = requestAnimationFrame(animate);
                analyser.getByteFrequencyData(dataArray);
                waveBars.forEach((bar, i) => {
                    const val = dataArray[Math.floor((i / waveBars.length) * bufferLength)] || 0;
                    bar.style.height = `${10 + (val / 255) * 30}px`;
                });
            }
            animate();
            waveBars.forEach(b => b.classList.remove('paused'));
 
            visualizerToggle.innerHTML = '<i class="fas fa-pause mr-1"></i> Stop';
            visualizerToggle.classList.add('bg-player-secondary');
            visualizerToggle.classList.remove('bg-player-primary');
        }
 
        function stopVisualizer() {
            if (animationId) { cancelAnimationFrame(animationId); animationId = null; }
            waveBars.forEach(b => {
                b.style.height = '20px';
                b.classList.add('paused');
            });
            visualizerToggle.innerHTML = '<i class="fas fa-play mr-1"></i> Start';
            visualizerToggle.classList.remove('bg-player-secondary');
            visualizerToggle.classList.add('bg-player-primary');
        }
 
        // ─── File Upload ──────────────────────────────────────────────────────────
        function handleFileUpload(files) {
            let addedCount = 0;
 
            Array.from(files).forEach(file => {
                if (!file.type.startsWith('audio/')) {
                    alert(`"${file.name}" is not an audio file.`);
                    return;
                }
                if (file.size > 50 * 1024 * 1024) {
                    alert(`"${file.name}" exceeds the 50 MB limit.`);
                    return;
                }
 
                // Create a persistent object URL (freed when playlist is cleared)
                const objectUrl = URL.createObjectURL(file);
                totalStorageBytes += file.size;
 
                const newSong = {
                    id: Date.now() + Math.random(),
                    title: file.name.replace(/\.[^/.]+$/, ''),
                    artist: 'Uploaded Track',
                    album: 'My Uploads',
                    duration: '--:--',          // placeholder until metadata loads
                    durationFormatted: '--:--',
                    durationSeconds: 0,
                    color: randomColor(),
                    lyrics: null,
                    src: objectUrl,
                    _objectUrl: objectUrl,
                    fileSize: file.size
                };
 
                // Read real duration via a temporary Audio element
                const tempAudio = new Audio();
                tempAudio.preload = 'metadata';
                tempAudio.src = objectUrl;
                tempAudio.addEventListener('loadedmetadata', () => {
                    const dur = tempAudio.duration;
                    newSong.durationSeconds = isFinite(dur) ? dur : 0;
                    newSong.durationFormatted = formatTime(dur);
                    newSong.duration = newSong.durationFormatted;
 
                    // If this is the currently playing song, update the displayed duration
                    if (currentSongIndex >= 0 && playlist[currentSongIndex] === newSong) {
                        totalDurationEl.textContent = newSong.durationFormatted;
                    }
                    updatePlaylist();
                    updateStats();
                });
 
                playlist.push(newSong);
                addedCount++;
            });
 
            if (addedCount > 0) {
                updatePlaylist();
                updateStats();
 
                // Auto-load the first uploaded song if nothing is playing
                if (currentSongIndex < 0) {
                    currentSongIndex = 0;
                    updateSongInfo();
                    updatePlaylist();
                }
 
                // Close the upload panel
                uploadAreaPanel.classList.add('hidden');
            }
        }
 
        // ─── Event Listeners ──────────────────────────────────────────────────────
        function initEventListeners() {
            // Player controls
            playPauseBtn.addEventListener('click', () => { initAudioContext(); togglePlayPause(); });
            prevBtn.addEventListener('click', playPrev);
            nextBtn.addEventListener('click', playNext);
            shuffleBtn.addEventListener('click', toggleShuffle);
            repeatBtn.addEventListener('click', toggleRepeat);
 
            // Progress & volume
            progressBar.addEventListener('input', seekAudio);
            volumeSlider.addEventListener('input', updateVolume);
 
            // Audio events
            audioPlayer.addEventListener('timeupdate', updateProgressBar);
 
            audioPlayer.addEventListener('loadedmetadata', () => {
                const dur = audioPlayer.duration;
                totalDurationEl.textContent = formatTime(dur);
 
                // Also patch the song object so playlist shows real duration
                if (currentSongIndex >= 0 && playlist[currentSongIndex]) {
                    const song = playlist[currentSongIndex];
                    if (isFinite(dur)) {
                        song.durationSeconds = dur;
                        song.durationFormatted = formatTime(dur);
                        song.duration = song.durationFormatted;
                        updatePlaylist();
                        updateStats();
                    }
                }
            });
 
            audioPlayer.addEventListener('ended', () => {
                if (isRepeating) {
                    audioPlayer.currentTime = 0;
                    audioPlayer.play();
                } else if (isAutoplay || true) {
                    // Auto-advance to next song by default
                    const next = (currentSongIndex + 1) % playlist.length;
                    if (next === 0 && !isAutoplay) {
                        // Reached end of playlist, stop
                        isPlaying = false;
                        setPlayingUI(false);
                        stopVisualizer();
                    } else {
                        playSong(next);
                    }
                } else {
                    isPlaying = false;
                    setPlayingUI(false);
                    stopVisualizer();
                }
            });
 
            audioPlayer.addEventListener('error', (e) => {
                console.warn('Audio error:', e);
            });
 
            // Upload panel
            uploadBtn.addEventListener('click', () => {
                uploadAreaPanel.classList.remove('hidden');
            });
            closeUploadBtn.addEventListener('click', () => {
                uploadAreaPanel.classList.add('hidden');
            });
 
            // Drop zone clicks open file picker
            dropZone.addEventListener('click', () => fileInput.click());
 
            dropZone.addEventListener('dragover', e => {
                e.preventDefault();
                dropZone.classList.add('dragover');
            });
            dropZone.addEventListener('dragleave', () => {
                dropZone.classList.remove('dragover');
            });
            dropZone.addEventListener('drop', e => {
                e.preventDefault();
                dropZone.classList.remove('dragover');
                handleFileUpload(e.dataTransfer.files);
            });
 
            fileInput.addEventListener('change', e => {
                handleFileUpload(e.target.files);
                fileInput.value = '';   // reset so same file can be re-uploaded
            });
 
            // Playlist controls
            clearPlaylistBtn.addEventListener('click', () => {
                if (!confirm('Clear entire playlist?')) return;
                audioPlayer.pause();
                audioPlayer.src = '';
 
                // Revoke all object URLs
                playlist.forEach(s => { if (s._objectUrl) URL.revokeObjectURL(s._objectUrl); });
 
                playlist = [];
                currentSongIndex = -1;
                totalStorageBytes = 0;
                isPlaying = false;
                setPlayingUI(false);
                stopVisualizer();
                updateSongInfo();
                updatePlaylist();
            });
 
            playAllBtn.addEventListener('click', () => {
                if (playlist.length > 0) playSong(0);
            });
 
            shufflePlaylistBtn.addEventListener('click', () => {
                if (playlist.length < 2) return;
                for (let i = playlist.length - 1; i > 0; i--) {
                    const j = Math.floor(Math.random() * (i + 1));
                    [playlist[i], playlist[j]] = [playlist[j], playlist[i]];
                }
                currentSongIndex = 0;
                updatePlaylist();
            });
 
            // Autoplay toggle
            autoplayToggle.addEventListener('click', () => {
                isAutoplay = !isAutoplay;
                autoplayToggle.classList.toggle('bg-player-primary/20', isAutoplay);
                const icon = autoplayToggle.querySelector('i');
                icon.className = isAutoplay ? 'fas fa-infinity text-player-primary mr-1' : 'fas fa-infinity mr-1';
            });
 
            // Lyrics modal
            lyricsBtn.addEventListener('click', () => {
                lyricsModal.classList.remove('hidden');
                lyricsModal.style.display = 'flex';
            });
            closeLyricsBtn.addEventListener('click', () => {
                lyricsModal.classList.add('hidden');
                lyricsModal.style.display = '';
            });
            lyricsModal.addEventListener('click', e => {
                if (e.target === lyricsModal) {
                    lyricsModal.classList.add('hidden');
                    lyricsModal.style.display = '';
                }
            });
 
            // Visualizer toggle
            visualizerToggle.addEventListener('click', () => {
                initAudioContext();
                if (animationId) {
                    stopVisualizer();
                } else {
                    startVisualizer();
                }
            });
 
            // Theme toggle
            themeToggle.addEventListener('click', () => {
                const icon = themeToggle.querySelector('i');
                if (document.documentElement.classList.toggle('dark')) {
                    icon.className = 'fas fa-sun';
                } else {
                    icon.className = 'fas fa-moon';
                }
            });
 
            // Playback speed
            speedSlider.addEventListener('input', () => {
                audioPlayer.playbackRate = speedSlider.value;
                playbackSpeedEl.textContent = `${parseFloat(speedSlider.value).toFixed(1)}x`;
            });
        }
 
        // ─── Init ─────────────────────────────────────────────────────────────────
        function init() {
            audioPlayer.volume = volumeSlider.value / 100;
            updateSongInfo();
            updatePlaylist();
            updateStats();
            initEventListeners();
        }
 
        document.addEventListener('DOMContentLoaded', init);
    </script>
